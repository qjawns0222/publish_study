🎨 스타일 가이드 (이 예제 전용)

색상 팔레트

  • 막대1: #4F46E5 - Primary
  • 막대2: #10B981 - Green
  • 막대3: #F59E0B - Orange

타이포그래피

  • 라벨: 12px, sans-serif

간격

  • 캔버스 크기: 600x400

💡 단계별 구현 가이드

1단계: HTML

<div class="chart-container">
  <canvas id="myChart" width="600" height="400"></canvas>
  <div class="legend" id="legend"></div>
</div>

2단계: JavaScript - Bar Chart

const canvas = document.getElementById('myChart');
const ctx = canvas.getContext('2d');

const data = [
  { label: '1월', value: 65, color: '#4F46E5' },
  { label: '2월', value: 59, color: '#10B981' },
  { label: '3월', value: 80, color: '#F59E0B' },
  { label: '4월', value: 81, color: '#EF4444' },
  { label: '5월', value: 56, color: '#8B5CF6' }
];

const padding = 40;
const chartWidth = canvas.width - padding * 2;
const chartHeight = canvas.height - padding * 2;
const barWidth = chartWidth / data.length - 10;
const maxValue = Math.max(...data.map(d => d.value));

// 배경
ctx.fillStyle = '#F9FAFB';
ctx.fillRect(0, 0, canvas.width, canvas.height);

// 막대 그리기
data.forEach((item, i) => {
  const x = padding + i * (barWidth + 10);
  const barHeight = (item.value / maxValue) * chartHeight;
  const y = canvas.height - padding - barHeight;

  // 막대
  ctx.fillStyle = item.color;
  ctx.fillRect(x, y, barWidth, barHeight);

  // 라벨
  ctx.fillStyle = '#222';
  ctx.font = '12px sans-serif';
  ctx.textAlign = 'center';
  ctx.fillText(item.label, x + barWidth / 2, canvas.height - padding + 20);

  // 값
  ctx.fillText(item.value, x + barWidth / 2, y - 10);
});

// 축
ctx.strokeStyle = '#D1D5DB';
ctx.lineWidth = 2;
ctx.beginPath();
ctx.moveTo(padding, padding);
ctx.lineTo(padding, canvas.height - padding);
ctx.lineTo(canvas.width - padding, canvas.height - padding);
ctx.stroke();
